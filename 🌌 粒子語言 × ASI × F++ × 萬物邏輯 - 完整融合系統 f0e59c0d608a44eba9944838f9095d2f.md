# Part I: 粒子語言系統

## 1. 語法定義

### EBNF 語法規範

```ebnf
program     ::= particle* expression*
particle    ::= 'particle' TYPE '{' property* '}'
property    ::= '@' IDENT ':' value
value       ::= STRING | NUMBER | vector | function
vector      ::= '(' NUMBER ',' NUMBER ')' | '(' NUMBER ',' NUMBER ',' NUMBER ')'
function    ::= IDENT '(' args ')'

expression  ::= term (OPERATOR term)*
term        ::= particle | IDENT | '(' expression ')'
OPERATOR    ::= '⊕' | '⊗' | '|>' | '>>='
```

## 2. 編譯器實現

```tsx
class ParticleCompiler {
  private tokens: Token[] = []
  private ast: AST | null = null
  private semanticTree: SemanticTree | null = null
  
  // 詞法分析
  tokenize(source: string): Token[] {
    const patterns = {
      PARTICLE: /particle/,
      TYPE: /[A-Z][A-Z0-9_]*/,
      PROPERTY: /@[a-z_]+/,
      OPERATOR: /⊕|⊗|\|>|>>=/,
      NUMBER: /[0-9]+\.?[0-9]*/,
      STRING: /"[^"]*"/,
      LPAREN: /\(/,
      RPAREN: /\)/,
      LBRACE: /\{/,
      RBRACE: /\}/,
      COLON: /:/,
      COMMA: /,/
    }
    
    this.tokens = []
    let pos = 0
    
    while (pos < source.length) {
      // 跳過空白
      if (/\s/.test(source[pos])) {
        pos++
        continue
      }
      
      let matched = false
      for (const [type, pattern] of Object.entries(patterns)) {
        const regex = new RegExp('^' + pattern.source)
        const match = source.slice(pos).match(regex)
        
        if (match) {
          this.tokens.push({
            type,
            value: match[0],
            pos,
            line: this.getLineNumber(source, pos)
          })
          pos += match[0].length
          matched = true
          break
        }
      }
      
      if (!matched) {
        throw new CompileError(`Unexpected character '${source[pos]}' at position ${pos}`)
      }
    }
    
    return this.tokens
  }
  
  // 語法分析
  parse(tokens: Token[]): AST {
    this.ast = {
      type: 'Program',
      particles: [],
      expressions: [],
      origin_signature: 'MrLiouWord'
    }
    
    let i = 0
    while (i < tokens.length) {
      if (tokens[i].type === 'PARTICLE') {
        const { node, nextIndex } = this.parseParticle(tokens, i)
        this.ast.particles.push(node)
        i = nextIndex
      } else {
        const { node, nextIndex } = this.parseExpression(tokens, i)
        this.ast.expressions.push(node)
        i = nextIndex
      }
    }
    
    return this.ast
  }
  
  // 語義分析
  analyze(ast: AST): SemanticTree {
    // 類型檢查
    this.typeCheck(ast)
    
    // 層級驗證
    this.verifyLayers(ast)
    
    // 簽名驗證
    if (ast.origin_signature !== 'MrLiouWord') {
      throw new SemanticError('Invalid origin signature')
    }
    
    // 構建語義樹
    this.semanticTree = this.buildSemanticTree(ast)
    return this.semanticTree
  }
  
  // 代碼生成
  codegen(tree: SemanticTree): ByteCode {
    const bytecode: ByteCode = {
      version: '2.0',
      instructions: [],
      metadata: {
        origin_signature: 'MrLiouWord',
        compiled_at: Date.now()
      }
    }
    
    for (const node of tree.nodes) {
      switch (node.type) {
        case 'ParticleDecl':
          bytecode.instructions.push({
            op: 'CREATE_PARTICLE',
            type: node.particleType,
            properties: node.properties,
            layer: node.layer
          })
          break
          
        case 'Combine':
          bytecode.instructions.push({
            op: 'COMBINE',
            operator: node.operator, // ⊕ or ⊗
            operands: node.operands
          })
          break
          
        case 'Transform':
          bytecode.instructions.push({
            op: 'TRANSFORM',
            particle: node.particle,
            targetLayer: node.targetLayer
          })
          break
          
        case 'Execute':
          bytecode.instructions.push({
            op: 'EXECUTE',
            particle: node.particle
          })
          break
      }
    }
    
    return bytecode
  }
  
  // 完整編譯流程
  compile(source: string): ExecutableCode {
    try {
      const tokens = this.tokenize(source)
      const ast = this.parse(tokens)
      const tree = this.analyze(ast)
      const bytecode = this.codegen(tree)
      
      return {
        bytecode,
        success: true,
        errors: []
      }
    } catch (error) {
      return {
        bytecode: null,
        success: false,
        errors: [error]
      }
    }
  }
}
```

## 3. 運行時實現

```tsx
class ParticleRuntime {
  private particles = new Map<string, Particle>()
  private stack: any[] = []
  private memory: Memory
  private asi: ASIEngine
  
  constructor() {
    this.memory = new Memory()
    this.asi = new ASIEngine()
  }
  
  // 執行字節碼
  async execute(bytecode: ByteCode): Promise<ExecutionResult> {
    for (const instruction of bytecode.instructions) {
      switch (instruction.op) {
        case 'CREATE_PARTICLE':
          await this.createParticle(instruction)
          break
          
        case 'COMBINE':
          await this.combineParticles(instruction)
          break
          
        case 'TRANSFORM':
          await this.transformParticle(instruction)
          break
          
        case 'EXECUTE':
          await this.executeParticle(instruction)
          break
      }
    }
    
    return {
      success: true,
      result: this.stack.pop(),
      origin_signature: 'MrLiouWord'
    }
  }
  
  private async createParticle(instruction: Instruction) {
    const particle: Particle = {
      id: generateId(),
      type: instruction.type,
      properties: instruction.properties,
      layer: instruction.layer,
      created_at: Date.now(),
      origin_signature: 'MrLiouWord'
    }
    
    this.particles.set(particle.id, particle)
    this.stack.push(particle.id)
  }
  
  private async combineParticles(instruction: Instruction) {
    const operands = instruction.operands.map(id => this.particles.get(id))
    
    let result: Particle
    if (instruction.operator === '⊕') {
      // 組合運算
      result = this.combine(operands)
    } else if (instruction.operator === '⊗') {
      // 張量積運算
      result = this.tensorProduct(operands)
    }
    
    this.particles.set(result.id, result)
    this.stack.push(result.id)
  }
  
  private combine(particles: Particle[]): Particle {
    // 實現 40-byte atom 組合邏輯
    return {
      id: generateId(),
      type: particles.map(p => p.type).reduce((a, b) => a | b),
      layer: Math.max(...particles.map(p => p.layer)),
      timestamp: Date.now(),
      origin_signature: 'MrLiouWord'
    }
  }
}
```

```tsx
class ParticleMonad implements Monad<Particle> {
  // pure :: a -> Particle a
  pure<T>(value: T): Particle<T> {
    return {
      type: 0x01, // CTX
      flags: 0x01, // ACTIVE  
      layer: 2,
      timestamp: Date.now() / 1000,
      position: { lat: 0, lon: 0 },
      data: 0n,
      simhash: 0n,
      value,
      origin_signature: 'MrLiouWord'
    }
  }
  
  // map :: (a -> b) -> Particle a -> Particle b
  map<A, B>(f: (a: A) => B, pa: Particle<A>): Particle<B> {
    return {
      ...pa,
      value: f(pa.value),
      timestamp: Date.now() / 1000,
      simhash: recomputeHash(pa)
    }
  }
  
  // flatMap :: (a -> Particle b) -> Particle a -> Particle b  
  flatMap<A, B>(f: (a: A) => Particle<B>, pa: Particle<A>): Particle<B> {
    const pb = f(pa.value)
    return this.combine(pa, pb)
  }
  
  // ap :: Particle (a -> b) -> Particle a -> Particle b
  ap<A, B>(pf: Particle<(a: A) => B>, pa: Particle<A>): Particle<B> {
    return this.map(pf.value, pa)
  }
  
  private combine<A, B>(pa: Particle<A>, pb: Particle<B>): Particle<B> {
    return {
      type: pa.type | pb.type,
      flags: pa.flags | pb.flags,
      layer: Math.max(pa.layer, pb.layer),
      timestamp: Math.max(pa.timestamp, pb.timestamp),
      position: {
        lat: (pa.position.lat + pb.position.lat) / 2,
        lon: (pa.position.lon + pb.position.lon) / 2
      },
      data: pa.data ^ pb.data,
      simhash: combineHash(pa.simhash, pb.simhash),
      value: pb.value,
      origin_signature: 'MrLiouWord'
    }
  }
}
```

```tsx
class L6_Cognition {
  private policy: PolicyNetwork
  private guard: GuardSystem
  private memory: CognitiveMemory
  
  // 策略決策
  async decide(
    ctx: Context,
    branches: Branch[],
    health: HealthReport
  ): Promise<Decision> {
    // 1. 應用策略
    const governed = await this.policy.apply(ctx, branches)
    
    // 2. 排序分支
    const ranked = governed.sort((a, b) => b.weight - a.weight)
    
    // 3. 選擇最優
    const picked = ranked[0]
    
    // 4. 守衛檢查
    const decision = await this.guard.lockdown(picked, health)
    
    // 5. 記錄決策
    await this.memory.record(decision)
    
    return decision
  }
  
  // 自我反思
  async reflect(): Promise<Insight> {
    const observation = await this.observe()
    const analysis = await this.analyze(observation)
    const insight = await this.synthesize(analysis)
    
    // 更新認知模型
    await this.updateCognitiveModel(insight)
    
    return insight
  }
```
