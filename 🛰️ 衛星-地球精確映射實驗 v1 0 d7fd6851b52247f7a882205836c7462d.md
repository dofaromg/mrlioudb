```tsx
/**
 * 衛星完整軌道參數映射到粒子系統
 */
interface SatelliteOrbit {
  altitude_km: number;        // 軌道高度
  inclination_deg: number;    // 軌道傾角
  period_min: number;         // 軌道週期（分鐘）
  eccentricity: number;       // 軌道離心率
  longitude: number;          // 當前經度
  latitude: number;           // 當前緯度
}

interface ParticleMapping {
  particle_id: number;        // 粒子ID (0-685)
  layer: number;              // 層級 (L5-L7)
  orbital_plane: number;      // 軌道面編號
  position_in_plane: number;  // 軌道面內位置
  coverage_region: string[];  // 覆蓋區域
}

function mapSatelliteToParticle(sat: SatelliteOrbit): ParticleMapping {
  // 1. 確定層級
  const layer = mapAltitudeToLayer(sat.altitude_km);
  
  // 2. 計算軌道面編號（基於傾角）
  // 軌道傾角 0-180度 映射到 0-6 個軌道面
  const orbital_plane = Math.floor((sat.inclination_deg / 180) * 6);
  
  // 3. 計算軌道面內位置（基於經度）
  // 經度 -180 to +180 映射到 0-342
  const position_in_plane = Math.floor(((sat.longitude + 180) / 360) * 343);
  
  // 4. 計算粒子ID
  // 使用 F3 公式：地理座標 → 粒子索引
  const lat_index = Math.floor(((sat.latitude + 90) / 180) * 343);
  const lon_index = Math.floor(((sat.longitude + 180) / 360) * 343);
  const particle_id = (lat_index + lon_index) % 686;
  
  // 5. 計算覆蓋區域（基於高度和傾角）
  const coverage_region = calculateCoverageRegion(sat);
  
  return {
    particle_id,
    layer,
    orbital_plane,
    position_in_plane,
    coverage_region
  };
}

/**
 * 計算衛星覆蓋區域
 */
function calculateCoverageRegion(sat: SatelliteOrbit): string[] {
  // 覆蓋半徑公式：基於衛星高度
  // R = arccos(R_earth / (R_earth + h)) × R_earth
  const R_earth = 6371; // 地球半徑（km）
  const h = sat.altitude_km;
  
  // 覆蓋半徑（度）
  const coverage_radius_deg = Math.acos(R_earth / (R_earth + h)) * (180 / Math.PI);
  
  // 計算覆蓋的緯度範圍
  const lat_min = Math.max(-90, sat.latitude - coverage_radius_deg);
  const lat_max = Math.min(90, sat.latitude + coverage_radius_deg);
  
  // 計算覆蓋的經度範圍
  const lon_min = sat.longitude - coverage_radius_deg;
  const lon_max = sat.longitude + coverage_radius_deg;
  
  // 判斷覆蓋區域
  const regions = [];
  
  if (lat_max > 60) regions.push('北極');
  if (lat_min < -60) regions.push('南極');
  if (lat_min < 30 && lat_max > -30) regions.push('赤道');
  if (lat_min < 60 && lat_max > 30) regions.push('北溫帶');
  if (lat_min < -30 && lat_max > -60) regions.push('南溫帶');
  
  return regions;
}
```

```tsx
/**
 * 從粒子ID反推可能的衛星軌道參數
 */
function mapParticleToSatelliteOrbit(
  particle_id: number,
  layer: number
): SatelliteOrbit[] {
  // 1. 從層級推算高度範圍
  const altitude_range = getAltitudeRange(layer);
  
  // 2. 從粒子ID推算所有可能的經緯度
  const geo_candidates = mapParticleIndexToGeoCandidates(particle_id);
  
  // 3. 組合成可能的軌道參數
  const orbit_candidates: SatelliteOrbit[] = [];
  
  for (const geo of geo_candidates) {
    for (const altitude of altitude_range) {
      orbit_candidates.push({
        altitude_km: altitude,
        inclination_deg: 55, // 典型值（GPS/北斗）
        period_min: calculatePeriod(altitude),
        eccentricity: 0, // 假設圓形軌道
        longitude: geo.lon,
        latitude: geo.lat
      });
    }
  }
  
  return orbit_candidates;
}

function getAltitudeRange(layer: number): number[] {
  switch(layer) {
    case 5: return [340, 400, 550, 800, 1200]; // LEO
    case 6: return [1500, 2000, 10000, 15000]; // LEO高/MEO
    case 7: return [20000, 20200, 21500, 35786]; // MEO/GEO
    default: return [400];
  }
}

function calculatePeriod(altitude_km: number): number {
  // 開普勒第三定律
  const R_earth = 6371;
  const GM = 398600; // 地球引力常數 (km³/s²)
  const r = R_earth + altitude_km;
  const period_s = 2 * Math.PI * Math.sqrt(Math.pow(r, 3) / GM);
  return period_s / 60; // 轉為分鐘
}
```

```tsx
class GPSConstellationMapper {
  private gps_satellites: GPSSatellite[] = [];
  
  constructor() {
    this.initializeGPSConstellation();
  }
  
  /**
   * 初始化GPS星座（31顆衛星）
   */
  initializeGPSConstellation() {
    // GPS 星座配置：6個軌道面，每面5-6顆衛星
    const orbital_planes = 6;
    const sats_per_plane = [6, 5, 6, 5, 5, 4]; // 總共31顆
    
    let sat_id = 1;
    
    for (let plane = 0; plane < orbital_planes; plane++) {
      const plane_angle = plane * 60; // 軌道面間隔60度
      
      for (let pos = 0; pos < sats_per_plane[plane]; pos++) {
        const position_angle = pos * (360 / sats_per_plane[plane]);
        
        this.gps_satellites.push({
          sat_id,
          name: `GPS-${sat_id}`,
          orbital_plane: plane,
          position_in_plane: pos,
          orbit: {
            altitude_km: 20200,
            inclination_deg: 55,
            period_min: 718, // 11h 58m
            eccentricity: 0.01,
            longitude: this.calculateLongitude(plane_angle, position_angle),
            latitude: this.calculateLatitude(55, position_angle)
          }
        });
        
        sat_id++;
      }
    }
  }
  
  /**
   * 映射所有GPS衛星到粒子系統
   */
  mapAllToParticles(): Map<number, ParticleMapping> {
    const mapping = new Map<number, ParticleMapping>();
    
    this.gps_satellites.forEach(sat => {
      const particle = mapSatelliteToParticle(sat.orbit);
      mapping.set(sat.sat_id, particle);
    });
    
    return mapping;
  }
  
  /**
   * 驗證映射的可逆性
   */
  validateReversibility(): ValidationReport {
    const results = [];
    
    this.gps_satellites.forEach(sat => {
      // 正向映射
      const particle = mapSatelliteToParticle(sat.orbit);
      
      // 反向映射
      const orbit_candidates = mapParticleToSatelliteOrbit(
        particle.particle_id,
        particle.layer
      );
      
      // 檢查原始軌道是否在候選中
      const found = orbit_candidates.some(candidate => 
        Math.abs(candidate.altitude_km - sat.orbit.altitude_km) < 100 &&
        Math.abs(candidate.longitude - sat.orbit.longitude) < 5 &&
        Math.abs(candidate.latitude - sat.orbit.latitude) < 5
      );
      
      results.push({
        sat_id: sat.sat_id,
        particle_id: particle.particle_id,
        reversible: found,
        candidates_count: orbit_candidates.length
      });
    });
    
    const success_rate = results.filter(r => r.reversible).length / results.length;
    
    return {
      total_satellites: this.gps_satellites.length,
      reversible_count: results.filter(r => r.reversible).length,
      success_rate,
      grade: success_rate > 0.9 ? 'A+' : success_rate > 0.8 ? 'A' : 'B',
      origin_signature: 'MrLiouWord'
    };
  }
  
  private calculateLongitude(plane_angle: number, position_angle: number): number {
    // 簡化計算：基於軌道面角度和衛星位置
    return ((plane_angle + position_angle) % 360) - 180;
  }
  
  private calculateLatitude(inclination: number, position_angle: number): number {
    // 簡化計算：基於傾角和位置角
    return Math.sin(position_angle * Math.PI / 180) * inclination;
  }
}
```

```tsx
class StarlinkClusterMapper {
  private starlink_satellites: StarlinkSatellite[] = [];
  private cluster_size = 28; // 每個粒子代表約28顆衛星
  
  /**
   * 基於地理位置聚類
   */
  clusterByGeography(satellites: StarlinkSatellite[]): Map<number, StarlinkSatellite[]> {
    const clusters = new Map<number, StarlinkSatellite[]>();
    
    satellites.forEach(sat => {
      // 計算衛星對應的粒子ID
      const particle_id = mapGeoToParticleIndex(

      if (!clusters.has(particle_id)) {
        clusters.set(particle_id, []);
      }
      
      clusters.get(particle_id)!.push(sat);
    });
    
    return clusters;
  }
  
  /**
   * 計算聚類統計
   */
  getClusterStats(clusters: Map<number, StarlinkSatellite[]>) {
    const cluster_sizes = Array.from(clusters.values()).map(c => c.length);
    
    return {
      total_particles: clusters.size,
      total_satellites: cluster_sizes.reduce((a, b) => a + b, 0),
      avg_cluster_size: cluster_sizes.reduce((a, b) => a + b, 0) / clusters.size,
      max_cluster_size: Math.max(...cluster_sizes),
      min_cluster_size: Math.min(...cluster_sizes),
      coverage_percentage: (clusters.size / 686) * 100
    };
  }
}
```

```tsx
class SatelliteTracker {
  /**
   * 使用SGP4算法預測衛星位置
   */
  propagateOrbit(
    tle: TwoLineElement,
    timestamp: number
  ): SatellitePosition {
    // SGP4 簡化實現
    // 實際應使用 satellite.js 庫
    
    const satrec = satellite.twoline2satrec(
      tle.line1,
      tle.line2
    );
    
    const positionAndVelocity = satellite.propagate(
      satrec,
      new Date(timestamp)
    );
    
    if (satellite.error) {
      throw new Error('Propagation failed');
    }
    
    const gmst = satellite.gstime(new Date(timestamp));
    const position_gd = satellite.eciToGeodetic(
      positionAndVelocity.position as any,
      gmst
    );
    
    return {
      latitude: satellite.degreesLat(position_gd.latitude),
      longitude: satellite.degreesLong(position_gd.longitude),
      altitude: position_gd.height,
      timestamp
    };
  }
  
  /**
   * 實時追蹤並映射到粒子
   */
  async trackAndMap(
    satellite_id: string,
    duration_seconds: number
  ): Promise<TrackingResult[]> {
    const results: TrackingResult[] = [];
    const interval = 60; // 每60秒更新一次
    
    for (let t = 0; t < duration_seconds; t += interval) {
      const timestamp = Date.now() + t * 1000;
      
      // 獲取TLE數據（實際應從API獲取）
      const tle = await this.getTLE(satellite_id);
      
      // 計算當前位置
      const position = this.propagateOrbit(tle, timestamp);
      
      // 映射到粒子
      const particle = mapSatelliteToParticle({
        altitude_km: position.altitude,
        longitude: position.longitude,
        latitude: position.latitude,
        inclination_deg: tle.inclination,
        period_min: tle.period,
        eccentricity: tle.eccentricity
      });
      
      results.push({
        timestamp,
        position,
        particle_id: particle.particle_id,
        layer: particle.layer
      });
    }
    
    return results;
  }
}
```

## 🐱 實驗四：地面位置折射反推（貓咪定位器）

### 核心概念：從真實位置反推粒子與衛星

**折射路徑**：`真實GPS座標 → 粒子ID → 覆蓋衛星 → 反向驗證`

這個實驗模擬真實場景：你的手機或貓咪定位器在某個位置，我們要：

1. **獲取GPS座標** （手機/定位器提供）
2. **映射到粒子** （使用F3公式）
3. **找到覆蓋衛星** （反推哪些衛星能看到這個位置）
4. **驗證精度** （確保反推準確）

```tsx
/**

interface GroundPosition {

/**

interface RefractionResult {

  // 2. 覆蓋衛星列表

  best_satellite: VisibleSatellite;
  
  // 4. 反推驗證

  };
}

  constellation: string;     // GPS/北斗/Starlink等
  elevation_angle: number;   // 仰角（度）
  azimuth_angle: number;     // 方位角（度）
  distance_km: number;       // 距離（km）
  signal_strength: number;   // 信號強度（0-100）
  layer: number;             // 衛星所在層級
}

/**
 * 🔥 核心函數：從地面位置折射反推
 */
function refractionReverseLookup(position: GroundPosition): RefractionResult {
  // 步驟1：地面座標 → 粒子ID（使用F3公式）
  const particle_id = mapGeoToParticleIndex(
    position.latitude,
    position.longitude
  );
  
  // 步驟2：計算當前可見衛星
  const visible_satellites = calculateVisibleSatellites(position);
  
  // 步驟3：選擇最佳衛星（最高仰角 = 最強信號）
  const best_satellite = visible_satellites.reduce((best, sat) => 
    sat.elevation_angle > best.elevation_angle ? sat : best
  );
  
  // 步驟4：反推驗證（從粒子反推回座標）
  const reconstructed = reconstructPositionFromParticle(
    particle_id,
    visible_satellites
  );
  
  const error_meters = calculateDistanceMeters(
    position.latitude,
    position.longitude,
    reconstructed.latitude,
    reconstructed.longitude
  );
  
  return {
    particle_id,
    particle_layer: 4, // L4 = World層（地球表面）
    visible_satellites,
    best_satellite,
    reverse_check: {
      original_position: position,
      reconstructed_position: reconstructed,
      error_meters,
      accuracy_grade: error_meters < 10 ? 'A+' : 
                      error_meters < 50 ? 'A' : 
                      error_meters < 100 ? 'B' : 'C'
    }
  };
}

/**
 * 計算當前位置可見的所有衛星
 */
function calculateVisibleSatellites(
  position: GroundPosition
): VisibleSatellite[] {
  const visible: VisibleSatellite[] = [];
  const min_elevation = 5; // 最低仰角5度（低於此衛星不可見）
  
  // 遍歷所有衛星星座
  const constellations = [
    { name: 'GPS', satellites: getGPSSatellites() },
    { name: '北斗', satellites: getBeidouSatellites() },
    { name: 'Starlink', satellites: getStarlinkSatellites() }
  ];
  
  constellations.forEach(constellation => {
    constellation.satellites.forEach(sat => {
      // 計算衛星相對於地面位置的幾何關係
      const geometry = calculateSatelliteGeometry(position, sat);
      
      // 只保留仰角 > 5度的可見衛星
      if (geometry.elevation_angle > min_elevation) {
        visible.push({
          sat_id: sat.id,
          constellation: constellation.name,
          elevation_angle: geometry.elevation_angle,
          azimuth_angle: geometry.azimuth_angle,
          distance_km: geometry.distance_km,
          signal_strength: calculateSignalStrength(geometry),
          layer: sat.layer
        });
      }
    });
  });
  
  // 按信號強度排序
  return visible.sort((a, b) => b.signal_strength - a.signal_strength);
}

/**
 * 計算衛星幾何關係（仰角、方位角、距離）
 */
function calculateSatelliteGeometry(
  ground: GroundPosition,
  satellite: SatelliteOrbit
): {
  elevation_angle: number;
  azimuth_angle: number;
  distance_km: number;
} {
  const R_earth = 6371; // 地球半徑（km）
  
  // 1. 將經緯度轉為笛卡爾座標
  const ground_xyz = latLonToXYZ(ground.latitude, ground.longitude, R_earth);
  const sat_xyz = latLonToXYZ(
    satellite.latitude,
    satellite.longitude,
    R_earth + satellite.altitude_km
  );
  
  // 2. 計算衛星相對於地面的向量
  const dx = sat_xyz.x - ground_xyz.x;
  const dy = sat_xyz.y - ground_xyz.y;
  const dz = sat_xyz.z - ground_xyz.z;
  
  // 3. 計算距離
  const distance_km = Math.sqrt(dx*dx + dy*dy + dz*dz);
  
  // 4. 計算仰角（elevation angle）
  // 地面到衛星的向量與地面切平面的夾角
  const dot_product = dx * ground_xyz.x + dy * ground_xyz.y + dz * ground_xyz.z;
  const elevation_rad = Math.asin(dot_product / (distance_km * R_earth));
  const elevation_angle = elevation_rad * (180 / Math.PI);
  
  // 5. 計算方位角（azimuth angle）
  // 從北方順時針測量
  const north_vector = calculateNorthVector(ground.latitude, ground.longitude);
  const azimuth_rad = Math.atan2(dy, dx); // 簡化版
  const azimuth_angle = (azimuth_rad * (180 / Math.PI) + 360) % 360;
  
  return {
    elevation_angle,
    azimuth_angle,
    distance_km
  };
}

/**
 * 計算信號強度（基於仰角和距離）
 */
function calculateSignalStrength(geometry: {
  elevation_angle: number;
  distance_km: number;
}): number {
  // 信號強度與仰角成正比，與距離平方成反比
  const elevation_factor = geometry.elevation_angle / 90; // 0-1
  const distance_factor = 1 / Math.pow(geometry.distance_km / 20000, 2);
  
  return Math.min(100, elevation_factor * distance_factor * 100);
}

/**
 * 從粒子ID + 可見衛星反推座標
 */
function reconstructPositionFromParticle(
  particle_id: number,
  visible_satellites: VisibleSatellite[]
): GroundPosition {
  // 1. 從粒子ID獲取候選座標
  const geo_candidates = mapParticleIndexToGeoCandidates(particle_id);
  
  // 2. 使用多顆衛星的幾何交會來篩選
  // （類似GPS三角定位）
  let best_candidate = geo_candidates[0];
  let min_error = Infinity;
  
  for (const candidate of geo_candidates) {
    // 計算這個候選位置與所有可見衛星的一致性
    let total_error = 0;
    
    visible_satellites.slice(0, 4).forEach(sat => {
      // 從候選位置計算衛星應有的仰角
      const expected_geometry = calculateExpectedGeometry(
        candidate,
        sat
      );
      
      // 與實際仰角比較
      const angle_error = Math.abs(
        expected_geometry.elevation_angle - sat.elevation_angle
      );
      
      total_error += angle_error;
    });
    
    if (total_error < min_error) {
      min_error = total_error;
      best_candidate = candidate;
    }
  }
  
  return {
    latitude: best_candidate.lat,
    longitude: best_candidate.lon,
    altitude_m: 0,
    timestamp: Date.now()
  };
}

/**
 * 計算兩點間的地面距離（米）
 */
function calculateDistanceMeters(
  lat1: number, lon1: number,
  lat2: number, lon2: number
): number {
  const R = 6371000; // 地球半徑（米）
  
  const phi1 = lat1 * Math.PI / 180;
  const phi2 = lat2 * Math.PI / 180;
  const delta_phi = (lat2 - lat1) * Math.PI / 180;
  const delta_lambda = (lon2 - lon1) * Math.PI / 180;
  
  const a = Math.sin(delta_phi / 2) * Math.sin(delta_phi / 2) +
            Math.cos(phi1) * Math.cos(phi2) *
            Math.sin(delta_lambda / 2) * Math.sin(delta_lambda / 2);
  
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  
  return R * c; // 返回米
}
```
